========
Match statement 
========

func prettyprint() {
    match this {
        case Fahrenheit(f) => {
            return "{0} F".format(f);
        }, 
        case Celsius(c) => {
            return "{0} C".format(c);
        }
    }
}

========
If statement
========

func prettyprint(a) {
    if a == 5 {
        println("a is 5");
    }
    
    if a == 5 {
        println("a is 5");
    } else {
        println("a is neither 5 nor 10");
    }
    
    if a == 5 {
        println("a is 5");
    } elsif a == 10 {
        println("a is 10");
    }
    
    if a == 5 {
        println("a is 5");
    } elsif a == 10 {
        println("a is 10");
    } else {
        println("a is neither 5 nor 10");
    }
}

    for i in [] {

        println(x);

        println("This list is empty!");

    while x > 0 {
        println("x is positive");
        x -= 1;
    } else {
        println("x is <= 0");
