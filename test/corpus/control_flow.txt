func prettyprint(a) {
    if a == 5 {
        println("a is 5");
    }
    
    if a == 5 {
        println("a is 5");
    } else {
        println("a is neither 5 nor 10");
    }
    
    if a == 5 {
        println("a is 5");
    } elsif a == 10 {
        println("a is 10");
    }
    
    if a == 5 {
        println("a is 5");
    } elsif a == 10 {
        println("a is 10");
    } else {
        println("a is neither 5 nor 10");
    }
}
